DBSCAN Clustering
Density-based clustering on sales data using sklearn
# Step 1: Import Libraries
from sklearn.cluster import DBSCAN
import pandas as pd
import matplotlib.pyplot as plt
from sklearn.preprocessing import StandardScaler

# Step 2: Load Dataset
mydf = pd.read_csv("D:/sales.csv")
print(mydf)

# Step 3: Convert columns into X
X = mydf[["annual_income", "spending_score"]]
print(X)

# Step 4: Scale your data using StandardScaler
X_scaled = StandardScaler().fit_transform(X)

# Step 5: Print scaled data
print(X_scaled)

# Step 6: Apply DBSCAN algorithm
model = DBSCAN(eps=0.5)
model.fit(X_scaled)

# Step 7: Find Labels
labels = model.fit_predict(X_scaled)

# Step 8: Print Labels and store into DataFrame
print(labels)
mydf['cluster'] = labels

# Step 9: Plot clusters in scatter plot
plt.scatter(mydf['annual_income'], mydf['spending_score'], c=mydf['cluster'], cmap='viridis')
plt.xlabel("Annual Income")
plt.ylabel("Spending Score")
plt.title("DBSCAN Clustering")
plt.colorbar(label='Cluster')
plt.show()

Circular Linked List (CLL)
Create, display, insert and delete nodes in a circular linked list
import java.util.*;

class Node {
    int data;
    Node next;
    Node(int ele) {
        data = ele;
        next = null;
    }
}

class cll {
    Node first = null;
    Node last  = null;

    public void create_cll() {
        int ele; char ans;
        Scanner ob3 = new Scanner(System.in);
        do {
            System.out.print("Enter Element: ");
            ele = ob3.nextInt();
            Node newnode = new Node(ele);
            if (first == null) first = newnode;
            else last.next = newnode;
            last = newnode;
            last.next = first;  // CIRCULAR: last always points to first
            System.out.print("Continue? (y): ");
            ans = ob3.next().charAt(0);
        } while (ans == 'y');
    }

    public void display() {
        Node q = first;
        if (first == null) return;
        do {
            System.out.print(q.data + "\t");
            q = q.next;
        } while (q != first);  // CIRCULAR: stop when back at first
        System.out.println();
    }

    public void insert_pos() {
        Scanner ob3 = new Scanner(System.in);
        System.out.print("Enter Position: "); int p = ob3.nextInt();
        System.out.print("Enter Element: ");  int ele = ob3.nextInt();
        Node newnode = new Node(ele);
        if (p == 1) {
            newnode.next = first; first = newnode; last.next = first;
        } else {
            Node q = first; int i = 1;
            while (q.next != first && i <= p - 2) { q = q.next; i++; }
            newnode.next = q.next; q.next = newnode;
            if (newnode.next == first) last = newnode;
        }
    }

    public void delete_pos() {
        Scanner ob3 = new Scanner(System.in);
        System.out.print("Enter Position: "); int p = ob3.nextInt();
        if (p == 1) {
            first = first.next; last.next = first;
        } else {
            Node q = first; int i = 1;
            while (q.next != first && i <= p - 2) { q = q.next; i++; }
            Node temp = q.next; q.next = temp.next;
            if (q.next == first) last = q;
        }
    }

    public static void main(String[] args) {
        Scanner ob1 = new Scanner(System.in);
        cll ob2 = new cll(); int ch;
        do {
            System.out.println("1. Create  2. Display  3. Insert  4. Delete  5. Exit");
            System.out.print("Choice: "); ch = ob1.nextInt();
            switch (ch) {
                case 1: ob2.create_cll(); break;
                case 2: ob2.display();    break;
                case 3: ob2.insert_pos(); break;
                case 4: ob2.delete_pos(); break;
                case 5: System.out.println("Bye bye"); break;
            }
        } while (ch != 5);
    }
}
